  unfolding F_map_comp o_assoc

  unfolding o_assoc[symmetric] ΣΣ9.map_comp0[symmetric]

  unfolding o_assoc[symmetric] ΣΣ9.map_comp0[symmetric] fst_convol ΣΣ9.map_id0 o_id

subsection{* The correspondence between coalgebras up to and coalgebras *}

(* This correspondence, and add9 the lemmas of this subsection, make

(* Coalgebra-up-to to coalgebra and vice versa: *)
definition cutΣΣ9Oc :: "('a => 'a ΣΣ9 F) => ('a ΣΣ9 => 'a ΣΣ9 F)"
where "cutΣΣ9Oc s ≡ F_map flat9 o dd9 o ΣΣ9_map <leaf9, s>"
definition cΣΣ9Ocut :: "('a ΣΣ9 => 'a ΣΣ9 F) => ('a => 'a ΣΣ9 F)"
where "cΣΣ9Ocut s' ≡ s' o leaf9"

lemma cΣΣ9Ocut_cutΣΣ9Oc: "cΣΣ9Ocut (cutΣΣ9Oc s) = s"
unfolding cΣΣ9Ocut_def cutΣΣ9Oc_def
unfolding o_assoc[symmetric] unfolding leaf9_natural
unfolding o_assoc apply(subst o_assoc[symmetric])
unfolding dd9_leaf9 unfolding o_assoc F_map_comp[symmetric] flat9_leaf9
unfolding F_map_id id_o by simp

lemma cutΣΣ9Oc_inj: "cutΣΣ9Oc s9 = cutΣΣ9Oc s2 <-> s9 = s2"
by (metis cΣΣ9Ocut_cutΣΣ9Oc)

lemma cΣΣ9Ocut_surj: "∃ s'. cΣΣ9Ocut s' = s"
using cΣΣ9Ocut_cutΣΣ9Oc by(rule exI[of _ "cutΣΣ9Oc s"])

(* Morphism-up-to to morphism and vice versa: *)
(* The dd9-extension of a function *)
definition extdd9 :: "('a => J) => ('a ΣΣ9 => J)"
where "extdd9 f ≡ eval9 o ΣΣ9_map f"
(* The restriction of a function *)  term eval9
definition restr :: "('a ΣΣ9 => J) => ('a => J)"
where "restr f' ≡ f' o leaf9"

(* We think of extdd9 and restr as operating
extdd9 : Hom_dd9(s,dtor_J) -> Hom(cutΣΣ9Oc s,dtor_J) and
restr : Hom(cutΣΣ9Oc s,dtor_J) -> Hom_dd9(s,dtor_J), where
Hom(cutΣΣ9Oc s,dtor_J) is the set of coalgebra morphisms betwee cutΣΣ9Oc s and dtor_J and
Hom_dd9(s,dtor_J) is the set of coalgebra up-to-dd9-morphisms between s and dtor_J  *)

(* extdd9 is wedd9-defined: *)
lemma extdd9_mor:
assumes f:  "F_map (extdd9 f) o s = dtor_J o f"
shows "F_map (extdd9 f) o cutΣΣ9Oc s = dtor_J o (extdd9 f)"
proof-
  have AA: "eval9 ** F_map eval9 o (ΣΣ9_map f ** F_map (ΣΣ9_map f) o <leaf9 , s>) =
            <f , F_map eval9 o (F_map (ΣΣ9_map f) o s)>"
  unfolding map_prod_o_convol unfolding leaf9_natural o_assoc eval9_leaf9 id_o  ..
  show ?thesis
  unfolding extdd9_def
  unfolding o_assoc eval9[symmetric]
  unfolding o_assoc[symmetric] ΣΣ9.map_comp0[symmetric]
  unfolding convol_comp[symmetric] id_o
  unfolding f[symmetric, unfolded extdd9_def]
  unfolding o_assoc
  apply(subst o_assoc[symmetric])
  unfolding F_map_comp o_assoc
  unfolding cutΣΣ9Oc_def
  unfolding o_assoc
  unfolding F_map_comp[symmetric] unfolding o_assoc[symmetric]
  unfolding flat9_natural[symmetric]
  unfolding o_assoc F_map_comp
  apply(subst o_assoc[symmetric], subst o_assoc[symmetric], subst o_assoc[symmetric])
  unfolding dd9_natural[symmetric]
  unfolding o_assoc unfolding F_map_comp[symmetric] eval9_flat9
  unfolding F_map_comp apply(subst o_assoc[symmetric])
  unfolding dd9_natural[symmetric] unfolding o_assoc[symmetric] ΣΣ9.map_comp0[symmetric]
  unfolding o_assoc[symmetric] AA[unfolded o_assoc[symmetric]] ..
qed

lemma mor_cutΣΣ9Oc_flat9:
assumes f': "F_map f' o cutΣΣ9Oc s = dtor_J o f'"
shows "eval9 o ΣΣ9_map f' = f' o flat9"
proof(rule trans) (* this proof is clearly improvable: *)
  def h ≡ "dd9 o ΣΣ9_map <id,cutΣΣ9Oc s>"
  have f'_id: "f' = f' o id" by simp
  show "eval9 o ΣΣ9_map f' = dtor_unfold_J h"
  apply(rule J.dtor_unfold_unique, rule sym)
  unfolding o_assoc eval9[symmetric]
  unfolding o_assoc[symmetric] ΣΣ9.map_comp0[symmetric]
  unfolding convol_comp_id1[symmetric] unfolding f'[symmetric]
  apply(subst f'_id)
  unfolding o_assoc ΣΣ9.map_comp0
  apply(subst o_assoc[symmetric])
  unfolding o_assoc[symmetric] F_map_comp
  unfolding h_def apply(rule sym)
  unfolding o_assoc apply(subst o_assoc[symmetric])
  unfolding dd9_natural[symmetric] unfolding o_assoc[symmetric]
  unfolding ΣΣ9.map_comp0[symmetric] map_prod_o_convol ..
  (*  *)
  have AA: "<id , cutΣΣ9Oc s> = (flat9 ** F_map flat9) o (id ** dd9) o <leaf9, ΣΣ9_map <leaf9 , s>>"
  unfolding map_prod_o_convol o_assoc map_prod.comp cutΣΣ9Oc_def o_id flat9_leaf9 ..
  have BB: "flat9 ** F_map flat9 o id ** dd9 o <leaf9 , ΣΣ9_map <leaf9 , s>> = flat9 ** F_map flat9 o id ** dd9 o <ΣΣ9_map leaf9 , ΣΣ9_map <leaf9 , s>>"
  unfolding map_prod.comp unfolding map_prod_o_convol unfolding o_id unfolding flat9_leaf9 leaf9_flat9 ..
  show "dtor_unfold_J h = f' o flat9"
  apply(rule J.dtor_unfold_unique[symmetric], rule sym)
  unfolding o_assoc f'[symmetric]
  unfolding F_map_comp o_assoc[symmetric]
  apply(rule arg_cong[of _ _ "op o (F_map f')"])
  unfolding h_def
  unfolding AA BB
  unfolding ΣΣ9.map_comp0 apply(rule sym)
  unfolding o_assoc apply(subst o_assoc[symmetric])
  unfolding dd9_natural
  unfolding o_assoc F_map_comp[symmetric]
  unfolding flat9_assoc unfolding F_map_comp
  unfolding cutΣΣ9Oc_def o_assoc[symmetric] apply(rule arg_cong[of _ _ "op o (F_map flat9)"])
  unfolding o_assoc
  unfolding o_assoc[symmetric] unfolding ΣΣ9.map_comp0[symmetric] unfolding map_prod_o_convol id_o
  unfolding flat9_natural[symmetric] unfolding o_assoc
  unfolding dd9_flat9[symmetric] unfolding o_assoc[symmetric] unfolding ΣΣ9.map_comp0[symmetric]
  unfolding convol_o unfolding ΣΣ9.map_comp0[symmetric] unfolding fst_convol ..

(* restr is wedd9-defined:  *)
lemma restr_mor:
assumes f': "F_map f' o cutΣΣ9Oc s = dtor_J o f'"
shows "F_map (extdd9 (restr f')) o s = dtor_J o restr f'"
unfolding extdd9_def restr_def ΣΣ9.map_comp0
unfolding o_assoc mor_cutΣΣ9Oc_flat9[OF f']
unfolding o_assoc[symmetric] leaf9_flat9 o_id
unfolding o_assoc f'[symmetric]
unfolding o_assoc[symmetric] cΣΣ9Ocut_cutΣΣ9Oc[unfolded cΣΣ9Ocut_def] ..

lemma extdd9_restr:
assumes f': "F_map f' o cutΣΣ9Oc s = dtor_J o f'"
shows "extdd9 (restr f') = f'"
proof-
  have "f' = eval9 o ΣΣ9_map f' o leaf9"
  unfolding o_assoc[symmetric] leaf9_natural
  unfolding o_assoc eval9_leaf9 by simp
  also have "... = eval9 o ΣΣ9_map (f' o leaf9)"
  unfolding ΣΣ9.map_comp0 o_assoc
  unfolding mor_cutΣΣ9Oc_flat9[OF f'] unfolding o_assoc[symmetric] flat9_leaf9 leaf9_flat9 ..
  finally have A: "f' = eval9 o ΣΣ9_map (f' o leaf9)" .
  show ?thesis unfolding extdd9_def restr_def A[symmetric] ..
qed

lemma restr_inj:
assumes f9': "F_map f9' o cutΣΣ9Oc s = dtor_J o f9'"
and f2': "F_map f2' o cutΣΣ9Oc s = dtor_J o f2'"
shows "restr f9' = restr f2' <-> f9' = f2'"
using extdd9_restr[OF f9'] extdd9_restr[OF f2'] by metis

lemma extdd9_surj:
assumes f': "F_map f' o cutΣΣ9Oc s = dtor_J o f'"
shows "∃ f. extdd9 f = f'"
using extdd9_restr[OF f'] by(rule exI[of _ "restr f'"])

lemma restr_extdd9:
assumes f: "F_map (extdd9 f) o s = dtor_J o f"
shows "restr (extdd9 f) = f"
proof-
  have "dtor_J o f = F_map (extdd9 f) o s" using assms unfolding extdd9_def by (rule sym)
  also have "... = dtor_J o restr (extdd9 f)"
  unfolding restr_def unfolding o_assoc extdd9_mor[OF f, symmetric]
  unfolding o_assoc[symmetric] cΣΣ9Ocut_cutΣΣ9Oc[unfolded cΣΣ9Ocut_def] ..
  finally have "dtor_J o f = dtor_J o restr (extdd9 f)" .
  thus ?thesis unfolding dtor_J_o_inj by (rule sym)
qed

lemma extdd9_inj:
assumes f1: "F_map (extdd9 f1) o s = dtor_J o f1"
and f2: "F_map (extdd9 f2) o s = dtor_J o f2"

  proof(rule trans)
    show "extdd9 f = dtor_unfold_J (cutΣΣ9Oc s)" apply(rule J.dtor_unfold_unique) using f .
    show "dtor_unfold_J (cutΣΣ9Oc s) = extdd9 (unfoldU9 s)"

  qed
  show ?thesis using A unfolding extdd9_inj[OF assms unfoldU9_pointfree] .

theorem coinductionU_genCngdd9:
assumes "∀ a b. R a b --> F_rel (genCngdd9 R) (dtor_J a) (dtor_J b)"
shows "R a b --> a = b"
proof-
  let ?R' = "genCngdd9 R"
  have "R ≤ Retr ?R'" using assms unfolding Retr_def[abs_def] by auto
  hence "R ≤ ?R' \<sqinter> Retr ?R'" using leq_genCngdd9 by auto
  moreover have "cngdd9 (?R' \<sqinter> Retr ?R')" using cngdd9_Retr[OF cngdd9_genCngdd9[of R]] .
  ultimately have "?R' ≤ ?R' \<sqinter> Retr ?R'" using genCngdd9_minimal by metis
  hence "?R' ≤ Retr ?R'" by auto
  hence "?R' a b --> a = b"  unfolding Retr_def[abs_def] by (intro J.dtor_coinduct) auto
  thus ?thesis using leq_genCngdd9 by auto
qed
